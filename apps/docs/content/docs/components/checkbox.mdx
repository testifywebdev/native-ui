## Usage

```tsx
import { Checkbox } from '@/components/ui/checkbox';
```

```tsx
const [checked, setChecked] = useState(false);

<Checkbox
  checked={checked}
  onCheckedChange={setChecked}
/>
```

## Examples

### With a label

```tsx
import { View, Text } from 'react-native';

<View style={{ flexDirection: 'row', alignItems: 'center', gap: 8 }}>
  <Checkbox
    checked={agreed}
    onCheckedChange={setAgreed}
    accessibilityLabel="Accept terms"
  />
  <Text>I agree to the Terms of Service</Text>
</View>
```

### Disabled state

```tsx
<Checkbox
  checked={true}
  onCheckedChange={() => {}}
  disabled
  accessibilityLabel="Read-only checkbox"
/>
```

### Inside a form list

```tsx
{options.map((option) => (
  <View key={option.id} style={{ flexDirection: 'row', alignItems: 'center', gap: 12, marginBottom: 12 }}>
    <Checkbox
      checked={selected.includes(option.id)}
      onCheckedChange={(val) => toggleOption(option.id, val)}
      accessibilityLabel={option.label}
    />
    <Text>{option.label}</Text>
  </View>
))}
```

## API Reference

### Checkbox

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `checked` | `boolean` | — | **Required.** Current checked state. |
| `onCheckedChange` | `(checked: boolean) => void` | — | **Required.** Callback when state changes. |
| `disabled` | `boolean` | `false` | Disables interaction and reduces opacity. |
| `accessibilityLabel` | `string` | `"Checkbox"` | Screen reader label. |
| `accessibilityHint` | `string` | `"Toggles the checkbox state"` | Screen reader hint. |
