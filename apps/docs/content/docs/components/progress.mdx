## Examples

### Static value

```tsx
<Progress value={33} />
<Progress value={66} />
<Progress value={100} />
```

### Animated on mount

```tsx
import { useState, useEffect } from 'react';

export function AnimatedProgress() {
  const [progress, setProgress] = useState(0);

  useEffect(() => {
    const timer = setTimeout(() => setProgress(75), 300);
    return () => clearTimeout(timer);
  }, []);

  return <Progress value={progress} />;
}
```

### Controlled by user action

```tsx
const [progress, setProgress] = useState(0);

<Progress value={progress} />
<Button title="Increment" onPress={() => setProgress((p) => Math.min(p + 10, 100))} />
<Button title="Reset" onPress={() => setProgress(0)} variant="outline" />
```

### Inside a card

```tsx
<Card>
  <CardHeader>
    <CardTitleGroup>
      <CardTitle>Uploading…</CardTitle>
      <CardDescription>{progress}% complete</CardDescription>
    </CardTitleGroup>
  </CardHeader>
  <CardContent>
    <Progress value={progress} />
  </CardContent>
</Card>
```

## API Reference

### Progress

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `number` | `0` | Current progress from `0` to `100`. Values outside this range are clamped automatically. |
| `style` | `ViewStyle` | — | Additional styles for the track container. |

> The progress bar uses a spring animation (`mass: 0.3, damping: 15, stiffness: 120`) for smooth, physical-feeling transitions between values.
>
> Accessibility: the root View has `accessibilityRole="progressbar"` with `accessibilityValue` min/max/now/text set automatically.
