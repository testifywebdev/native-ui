## Usage

```tsx
import { DatePicker } from '@/components/ui/date-picker';
```

```tsx
const [date, setDate] = useState('');

<DatePicker
  value={date}
  onChange={setDate}
/>
```

## Examples

### With a default value

```tsx
<DatePicker
  value="2024-06-15"
  onChange={(date) => console.log(date)}
/>
```

### Custom placeholder

```tsx
<DatePicker
  placeholder="Select your birthday"
  onChange={setDate}
/>
```

### Inside a form field

```tsx
import { Field, FieldLabel } from '@/components/ui/field';

<Field>
  <FieldLabel>Appointment Date</FieldLabel>
  <DatePicker
    value={appointmentDate}
    onChange={setAppointmentDate}
    placeholder="Choose a date"
  />
</Field>
```

### With custom container style

```tsx
<DatePicker
  value={date}
  onChange={setDate}
  style={{ borderRadius: 8, borderColor: '#8b5cf6' }}
/>
```

## API Reference

### DatePicker

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | — | Selected date as `YYYY-MM-DD` string. |
| `onChange` | `(date: string) => void` | — | Callback with the selected date string. |
| `placeholder` | `string` | `"Pick a date"` | Text shown when no date is selected. |
| `style` | `ViewStyle` | — | Additional styles for the trigger input row. |

> Dates are stored and passed as ISO date strings (`YYYY-MM-DD`) and displayed in `en-US` long format (e.g. "June 15, 2024").
