## Usage

```tsx
import {
  Card,
  CardHeader,
  CardTitleGroup,
  CardTitle,
  CardDescription,
  CardAction,
  CardContent,
  CardFooter,
} from '@/components/ui/card';
```

```tsx
<Card>
  <CardHeader>
    <CardTitleGroup>
      <CardTitle>My Card</CardTitle>
      <CardDescription>A brief description</CardDescription>
    </CardTitleGroup>
    <CardAction>
      {/* e.g. a menu button */}
    </CardAction>
  </CardHeader>
  <CardContent>
    {/* main content */}
  </CardContent>
  <CardFooter>
    <Button title="Action" />
  </CardFooter>
</Card>
```

## Examples

### Simple card

```tsx
<Card>
  <CardTitle>Notifications</CardTitle>
  <CardContent>
    <Text>You have 3 unread messages.</Text>
  </CardContent>
</Card>
```

### Compact size

```tsx
<Card size="sm">
  <CardContent>
    <Text>A compact card with less padding.</Text>
  </CardContent>
</Card>
```

### Stat card

```tsx
<Card>
  <CardHeader>
    <CardTitleGroup>
      <CardTitle>Total Revenue</CardTitle>
      <CardDescription>+20.1% from last month</CardDescription>
    </CardTitleGroup>
    <CardAction>
      <Feather name="trending-up" size={20} />
    </CardAction>
  </CardHeader>
  <CardContent>
    <Text style={{ fontSize: 32, fontWeight: '700' }}>$45,231.89</Text>
  </CardContent>
</Card>
```

## API Reference

### Card

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `"default" \| "sm"` | `"default"` | Padding and border-radius size. `"sm"` uses `spacing.md` / `radius.lg`; `"default"` uses `spacing.lg` / `radius.xl`. |
| `style` | `ViewStyle` | — | Additional container styles. |

### CardHeader

A row container that places its children between a `CardTitleGroup` and an optional `CardAction`.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles. |

### CardTitleGroup

Wraps `CardTitle` and `CardDescription` in a flex column.

### CardTitle

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `TextStyle` | — | Additional text styles. |

### CardDescription

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `TextStyle` | — | Additional text styles. |

### CardAction

Non-shrinkable slot for an action button or icon in the card header.

### CardContent

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles. |

### CardFooter

A right-aligned row for action buttons, separated from the content by a top border.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles. |
