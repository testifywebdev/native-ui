## Usage

```tsx
import Badge from '@/components/ui/badge';
```

```tsx
<Badge label="New" variant="default" />
```

## Examples

### All variants

```tsx
<Badge label="Default" variant="default" />
<Badge label="Secondary" variant="secondary" />
<Badge label="Destructive" variant="destructive" />
<Badge label="Outline" variant="outline" />
<Badge label="Ghost" variant="ghost" />
```

### With an icon

```tsx
import { Feather } from '@expo/vector-icons';

<Badge
  label="Verified"
  variant="default"
  icon={<Feather name="check-circle" size={12} color="white" />}
/>
```

### Status indicators

```tsx
<Badge label="Active" variant="default" />
<Badge label="Pending" variant="secondary" />
<Badge label="Failed" variant="destructive" />
<Badge label="Draft" variant="ghost" />
```

### With custom styles

```tsx
<Badge
  label="Premium"
  variant="outline"
  style={{ borderColor: '#f59e0b' }}
  textStyle={{ color: '#f59e0b' }}
/>
```

## API Reference

### Badge

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | — | **Required.** Text content of the badge. |
| `variant` | `"default" \| "secondary" \| "destructive" \| "outline" \| "ghost"` | `"default"` | Visual style variant. |
| `icon` | `ReactNode` | — | Optional icon rendered before the label. |
| `style` | `ViewStyle` | — | Additional container styles. |
| `textStyle` | `TextStyle` | — | Additional label text styles. |
| `accessibilityLabel` | `string` | `label` | Custom accessibility label. |

## Variant Reference

| Variant | Background | Text | Border |
|---------|-----------|------|--------|
| `default` | `primary` | `primaryForeground` | transparent |
| `secondary` | `secondary` | `secondaryForeground` | transparent |
| `destructive` | `destructive` | `destructiveForeground` | transparent |
| `outline` | transparent | `foreground` | `border` |
| `ghost` | `muted` | `mutedForeground` | transparent |
