```tsx
import Input from '@/components/ui/input';
```

```tsx
<Input
  label="Username"
  placeholder="Enter your username"
  value={username}
  onChangeText={setUsername}
/>
```

## Examples

### With icons

```tsx
import { Feather } from '@expo/vector-icons';

<Input
  label="Search"
  placeholder="Search..."
  leftIcon={<Feather name="search" size={16} color="#71717a" />}
/>
```

### Password input

```tsx
const [show, setShow] = useState(false);

<Input
  label="Password"
  placeholder="••••••••"
  secureTextEntry={!show}
  rightIcon={
    <Feather name={show ? 'eye-off' : 'eye'} size={16} color="#71717a" />
  }
  onRightIconPress={() => setShow((v) => !v)}
/>
```

### Error state

```tsx
<Input
  label="Email"
  placeholder="you@example.com"
  value={email}
  onChangeText={setEmail}
  error="Please enter a valid email address."
/>
```

### With hint text

```tsx
<Input
  label="Username"
  placeholder="Choose a username"
  hint="Must be 3–20 characters. Letters and numbers only."
/>
```

### Ghost variant

```tsx
<Input
  placeholder="Search..."
  variant="ghost"
  leftIcon={<Feather name="search" size={16} color="#71717a" />}
/>
```

### Multiline (textarea)

```tsx
<Input
  label="Bio"
  placeholder="Tell us about yourself..."
  multiline
  numberOfLines={4}
/>
```

### Disabled state

```tsx
<Input
  label="Read-only"
  value="Cannot change this"
  disabled
/>
```

## API Reference

### Input

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | — | Label text rendered above the input. |
| `placeholder` | `string` | — | Placeholder text. |
| `value` | `string` | — | Controlled value. |
| `onChangeText` | `(text: string) => void` | — | Change handler. |
| `variant` | `"default" \| "ghost" \| "destructive"` | `"default"` | Visual style. `"ghost"` removes the border at rest. |
| `disabled` | `boolean` | `false` | Disables the input and reduces opacity. |
| `error` | `string` | — | Error message rendered below the input. Turns the border destructive. |
| `hint` | `string` | — | Helper text rendered below the input (hidden when `error` is set). |
| `leftIcon` | `ReactNode` | — | Icon rendered inside the left edge. |
| `rightIcon` | `ReactNode` | — | Icon rendered inside the right edge. |
| `onRightIconPress` | `() => void` | — | Makes the right icon pressable (e.g., toggle password visibility). |
| `secureTextEntry` | `boolean` | — | Masks input text. |
| `inputMode` | `InputModeOptions` | — | Keyboard type hint (e.g., `"email"`, `"numeric"`). |
| `keyboardType` | `KeyboardTypeOptions` | — | Low-level keyboard type. |
| `multiline` | `boolean` | `false` | Enables multiline (textarea) mode. |
| `numberOfLines` | `number` | `1` | Number of visible lines in multiline mode. |
| `containerStyle` | `ViewStyle` | — | Outer wrapper styles. |
| `inputStyle` | `TextStyle` | — | Inner `TextInput` styles. |
| `labelStyle` | `TextStyle` | — | Label text styles. |
