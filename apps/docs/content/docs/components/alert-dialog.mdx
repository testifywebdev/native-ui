## Usage

```tsx
import {
  AlertDialog,
  AlertDialogTrigger,
  AlertDialogContent,
  AlertDialogHeader,
  AlertDialogFooter,
  AlertDialogTitle,
  AlertDialogDescription,
  AlertDialogAction,
  AlertDialogCancel,
} from '@/components/ui/alert-dialog';
```

```tsx
<AlertDialog>
  <AlertDialogTrigger asChild>
    <Button title="Open" />
  </AlertDialogTrigger>
  <AlertDialogContent>
    <AlertDialogHeader>
      <AlertDialogTitle>Confirm Action</AlertDialogTitle>
      <AlertDialogDescription>
        Are you sure you want to proceed?
      </AlertDialogDescription>
    </AlertDialogHeader>
    <AlertDialogFooter>
      <AlertDialogCancel asChild>
        <Button title="Cancel" variant="outline" />
      </AlertDialogCancel>
      <AlertDialogAction asChild>
        <Button title="Confirm" />
      </AlertDialogAction>
    </AlertDialogFooter>
  </AlertDialogContent>
</AlertDialog>
```

## Examples

### Controlled state

```tsx
import { useState } from 'react';

export function ControlledAlertDialog() {
  const [open, setOpen] = useState(false);

  return (
    <AlertDialog open={open} onOpenChange={setOpen}>
      <AlertDialogTrigger asChild>
        <Button title="Open Dialog" onPress={() => setOpen(true)} />
      </AlertDialogTrigger>
      <AlertDialogContent>
        <AlertDialogHeader>
          <AlertDialogTitle>Controlled Dialog</AlertDialogTitle>
          <AlertDialogDescription>
            This dialog is controlled externally.
          </AlertDialogDescription>
        </AlertDialogHeader>
        <AlertDialogFooter>
          <AlertDialogCancel asChild>
            <Button title="Cancel" variant="outline" onPress={() => setOpen(false)} />
          </AlertDialogCancel>
          <AlertDialogAction asChild>
            <Button title="OK" onPress={() => setOpen(false)} />
          </AlertDialogAction>
        </AlertDialogFooter>
      </AlertDialogContent>
    </AlertDialog>
  );
}
```

## API Reference

### AlertDialog

Root component. Manages open/closed state.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | — | Controlled open state. |
| `onOpenChange` | `(open: boolean) => void` | — | Callback when open state changes. |

### AlertDialogTrigger

The element that opens the dialog. Use `asChild` to compose with your own component.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | `false` | Merge props onto the child element. |

### AlertDialogContent

The dialog panel. Renders inside a portal with a backdrop overlay.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `portalHost` | `string` | — | Custom portal host name. |
| `style` | `ViewStyle` | — | Additional styles. |

### AlertDialogHeader

Container for `AlertDialogTitle` and `AlertDialogDescription`.

### AlertDialogFooter

Container for action buttons, laid out in a row aligned to the right.

### AlertDialogTitle

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `TextStyle` | — | Additional text styles. |

### AlertDialogDescription

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `TextStyle` | — | Additional text styles. |

### AlertDialogAction

Wraps the confirm action. Closes the dialog when pressed.

### AlertDialogCancel

Wraps the cancel action. Closes the dialog when pressed.
