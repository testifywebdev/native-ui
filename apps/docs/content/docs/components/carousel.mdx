## Usage

```tsx
import {
  Carousel,
  CarouselContent,
  CarouselItem,
  CarouselPrevious,
  CarouselNext,
  CarouselDots,
} from '@/components/ui/carousel';
```

```tsx
<Carousel>
  <CarouselContent>
    <CarouselItem>{/* slide content */}</CarouselItem>
    <CarouselItem>{/* slide content */}</CarouselItem>
  </CarouselContent>
  <CarouselPrevious />
  <CarouselNext />
</Carousel>
```

## Examples

### With autoplay

```tsx
<Carousel autoplayInterval={3000} loop>
  <CarouselContent>
    {items.map((item) => (
      <CarouselItem key={item.id}>
        <SlideCard item={item} />
      </CarouselItem>
    ))}
  </CarouselContent>
  <CarouselDots />
</Carousel>
```

### Vertical carousel

```tsx
<Carousel orientation="vertical">
  <CarouselContent>
    {items.map((item) => (
      <CarouselItem key={item.id} style={{ height: 200 }}>
        <SlideCard item={item} />
      </CarouselItem>
    ))}
  </CarouselContent>
</Carousel>
```

### Looping carousel

```tsx
<Carousel loop>
  <CarouselContent>
    <CarouselItem>
      <SlideA />
    </CarouselItem>
    <CarouselItem>
      <SlideB />
    </CarouselItem>
    <CarouselItem>
      <SlideC />
    </CarouselItem>
  </CarouselContent>
  <CarouselPrevious />
  <CarouselNext />
  <CarouselDots />
</Carousel>
```

## API Reference

### Carousel

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `loop` | `boolean` | `false` | Enables infinite looping. |
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | Scroll direction. |
| `autoplayInterval` | `number` | `0` | Milliseconds between auto-advances. `0` disables autoplay. |
| `style` | `ViewStyle` | — | Additional styles for the root container. |

### CarouselContent

The scrollable list container. Accepts all `FlatList` props except `data`, `renderItem`, `horizontal`, and `ref`.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles for the list wrapper. |

### CarouselItem

One slide in the carousel. Its size is automatically set to fill the carousel width (or height in vertical mode).

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles for the slide container. |

### CarouselPrevious

A button that scrolls to the previous slide. Automatically disabled on the first slide when `loop` is `false`.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles for the button. |

### CarouselNext

A button that scrolls to the next slide. Automatically disabled on the last slide when `loop` is `false`.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles for the button. |

### CarouselDots

Pagination dot indicators. Tapping a dot jumps to that slide.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles for the dots container. |
