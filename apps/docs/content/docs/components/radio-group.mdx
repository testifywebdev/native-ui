## Usage

```tsx
import { RadioGroup, RadioGroupItem } from '@/components/ui/radio-group';
```

```tsx
<RadioGroup value={value} onValueChange={setValue}>
  <RadioGroupItem value="option-a" />
  <RadioGroupItem value="option-b" />
</RadioGroup>
```

## Examples

### With labels

```tsx
import { View, Text } from 'react-native';

<RadioGroup value={selected} onValueChange={setSelected}>
  {['Option A', 'Option B', 'Option C'].map((label) => {
    const val = label.toLowerCase().replace(' ', '-');
    return (
      <View key={val} style={{ flexDirection: 'row', alignItems: 'center', gap: 10 }}>
        <RadioGroupItem value={val} aria-labelledby={`lbl-${val}`} />
        <Text nativeID={`lbl-${val}`}>{label}</Text>
      </View>
    );
  })}
</RadioGroup>
```

### Disabled items

```tsx
<RadioGroup value={selected} onValueChange={setSelected}>
  <RadioGroupItem value="enabled" />
  <RadioGroupItem value="disabled" disabled />
</RadioGroup>
```

### Horizontal layout

```tsx
<RadioGroup
  value={selected}
  onValueChange={setSelected}
  style={{ flexDirection: 'row', gap: 16 }}
>
  <RadioGroupItem value="yes" />
  <RadioGroupItem value="no" />
</RadioGroup>
```

### Inside a form with Field

```tsx
<Field>
  <FieldLabel>Preferred layout</FieldLabel>
  <RadioGroup value={layout} onValueChange={setLayout}>
    {['grid', 'list', 'compact'].map((val) => (
      <View key={val} style={{ flexDirection: 'row', alignItems: 'center', gap: 10 }}>
        <RadioGroupItem value={val} />
        <Text style={{ textTransform: 'capitalize' }}>{val}</Text>
      </View>
    ))}
  </RadioGroup>
</Field>
```

## API Reference

### RadioGroup

Extends `@rn-primitives/radio-group` Root props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | — | The currently selected item value. |
| `onValueChange` | `(value: string) => void` | — | Callback when selection changes. |
| `style` | `ViewStyle` | — | Additional container styles. Defaults to `gap: spacing.md`. |

### RadioGroupItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | — | **Required.** Unique value for this option. |
| `disabled` | `boolean` | `false` | Disables this item (reduces opacity, prevents selection). |
| `style` | `ViewStyle` | — | Additional styles for the outer circle. |
| `aria-labelledby` | `string` | — | `nativeID` of a sibling `Text` element used as the accessibility label. |
